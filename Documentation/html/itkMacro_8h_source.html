00683 //----------------------------------------------------------------------------
00684 // Setup legacy code policy.
00685 //
00686 // CMake options ITK_LEGACY_REMOVE and ITK_LEGACY_SILENT are converted
00687 // They may be used to completely remove legacy code or silence the
00688 // warnings.  The default is to warn about their use.
00689 //
00690 // Source files that test the legacy code may define ITK_LEGACY_TEST
00691 // like this:
00692 //
00693 //  #define ITK_LEGACY_TEST
00694 //  #include "itkClassWithDeprecatedMethod.h"
00695 //
00696 // in order to silence the warnings for calling deprecated methods.
00697 // No other source files in ITK should call the methods since they are
00698 // provided only for compatibility with older user code.
00699 
00700 // Define itkLegacyMacro to mark legacy methods where they are
00701 // declared in their class.  Example usage:
00702 //
00703 //   // @deprecated Replaced by MyOtherMethod() as of ITK 2.0.
00704 //   itkLegacyMacro(void MyMethod());
00705 #if defined( ITK_LEGACY_REMOVE )
00706 // Remove legacy methods completely.  Put a bogus declaration in
00707 // place to avoid stray semicolons because this is an error for some
00708 // compilers.  Using a class forward declaration allows any number
