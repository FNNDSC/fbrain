00657 void LabelFusionTool<T>::ComputePatchRegion(typename itkTImage::IndexType p, typename itkTImage::RegionType & imageRegion, typename itkTImage::RegionType & patchRegion)
00658 {
00659   //create an appropriate patch region around the current pixel p, and consider an offset for pixels close to boundaries.
00660   //the patch region is defined in the image coordinate system.
00661   //the offset allows to do the link between the patch coordinate system and the one of the image.
00662   typename itkTImage::RegionType::IndexType start;
00663   typename itkTImage::RegionType::IndexType offset;
00664   typename itkTImage::RegionType::SizeType size;
00665 
00666   for(unsigned int i=0; i!= p.GetIndexDimension(); i++){
00667     start[i] = p[i] - m_halfPatchSize[i];
00668     size[i] = m_fullPatchSize[i];
00669     offset[i] = 0;
00670     
00671     if(start[i] < 0){                  //if the starting index is outside the image (<0)
00672       size[i] += start[i];
00673       offset[i] = -start[i];
00674       start[i] = 0;
00675     }
00676     else if(start[i] >= (int)m_size[i]){    //if the starting index is outside the image (>image size)
00677       start[i] = m_size[i]-1;
00678       size[i] = 0;
00679     }
00680     
00681     int d = (start[i] + size[i]) - m_size[i]; //if the region is not fully inside the image
00682     if(d>0){
00683       size[i] = size[i] - d;
00684       if(size[i] < 0) size[i] = 0;
00685     }    
00686   }  
00687   imageRegion.SetSize( size );
00688   imageRegion.SetIndex( start );    
00689   patchRegion.SetSize( size );
00690   patchRegion.SetIndex( offset );
